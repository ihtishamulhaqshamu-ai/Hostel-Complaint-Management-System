    <script>
        // IMPORTANT NOTE: This system uses in-memory storage (JavaScript variables) instead of localStorage
        // because localStorage is not supported in Claude.ai artifacts.
        // When you deploy this to your own environment, you can uncomment the localStorage code below.

        let currentUser = null;
        let complaints = [];
        let complaintIdCounter = 1;

        // Initialize system
        function initSystem() {
            // Demo data
            complaints = [
                {
                    id: 'C001',
                    studentName: 'Ali Ahmed',
                    regNo: 'B25F0001',
                    roomNo: 'A-101',
                    type: 'Electrical Fault',
                    description: 'Fan not working',
                    contact: '0300-1234567',
                    status: 'Resolved',
                    date: '2024-12-10',
                    submittedBy: 'student'
                }
            ];
            complaintIdCounter = 2;
        }

        // Login functionality
        document.getElementById('loginForm').addEventListener('submit', function(e) {
            e.preventDefault();
            const username = document.getElementById('loginUsername').value;
            const password = document.getElementById('loginPassword').value;

            // Simple authentication (in real app, verify against database)
            if ((username === 'student' && password === 'student123') || 
                (username === 'admin' && password === 'admin123')) {
                currentUser = {
                    username: username,
                    role: username === 'admin' ? 'admin' : 'student'
                };
                
                document.getElementById('mainNav').style.display = 'block';
                document.getElementById('userDisplay').textContent = `Welcome, ${username}`;
                
                if (currentUser.role === 'admin') {
                    document.getElementById('adminLink').style.display = 'block';
                    showPage('admin');
                } else {
                    showPage('home');
                }
            } else {
                alert('Invalid credentials!');
            }
        });

        // Logout
        function logout() {
            currentUser = null;
            document.getElementById('mainNav').style.display = 'none';
            document.getElementById('adminLink').style.display = 'none';
            showPage('login');
            document.getElementById('loginForm').reset();
        }

        // Page navigation
        function showPage(pageId) {
            const pages = document.querySelectorAll('.page');
            pages.forEach(page => page.classList.remove('active'));
            document.getElementById(pageId).classList.add('active');

            if (pageId === 'status') {
                displayUserComplaints();
            } else if (pageId === 'admin') {
                displayAdminDashboard();
            }
        }

        // Submit complaint
        document.getElementById('complaintForm').addEventListener('submit', function(e) {
            e.preventDefault();
            
            const complaint = {
                id: 'C' + String(complaintIdCounter++).padStart(3, '0'),
                studentName: document.getElementById('studentName').value,
                regNo: document.getElementById('regNo').value,
                roomNo: document.getElementById('roomNo').value,
                type: document.getElementById('complaintType').value,
                description: document.getElementById('description').value,
                contact: document.getElementById('contactNo').value,
                status: 'Pending',
                date: new Date().toISOString().split('T')[0],
                submittedBy: currentUser.username
            };

            complaints.push(complaint);
            
            alert('Complaint submitted successfully! Complaint ID: ' + complaint.id);
            this.reset();
            showPage('status');
        });

        // Display user complaints
        function displayUserComplaints() {
            const userComplaints = complaints.filter(c => c.submittedBy === currentUser.username);
            const container = document.getElementById('userComplaintsList');
            
            if (userComplaints.length === 0) {
                container.innerHTML = '<p style="text-align: center; color: #666; padding: 2rem;">No complaints submitted yet.</p>';
                return;
            }

            let html = '<table class="status-table"><thead><tr><th>ID</th><th>Room</th><th>Type</th><th>Date</th><th>Status</th></tr></thead><tbody>';
            
            userComplaints.forEach(complaint => {
                const statusClass = complaint.status === 'Resolved' ? 'status-resolved' : 
                                   complaint.status === 'In Progress' ? 'status-progress' : 'status-pending';
                html += `
                    <tr>
                        <td>${complaint.id}</td>
                        <td>${complaint.roomNo}</td>
                        <td>${complaint.type}</td>
                        <td>${complaint.date}</td>
                        <td><span class="status-badge ${statusClass}">${complaint.status}</span></td>
                    </tr>
                `;
            });
            
            html += '</tbody></table>';
            container.innerHTML = html;
        }

        // Display admin dashboard
        function displayAdminDashboard() {
            // Update statistics
            document.getElementById('totalComplaints').textContent = complaints.length;
            document.getElementById('pendingComplaints').textContent = complaints.filter(c => c.status === 'Pending').length;
            document.getElementById('progressComplaints').textContent = complaints.filter(c => c.status === 'In Progress').length;
            document.getElementById('resolvedComplaints').textContent = complaints.filter(c => c.status === 'Resolved').length;

            // Display all complaints
            const container = document.getElementById('adminComplaintsList');
            
            if (complaints.length === 0) {
                container.innerHTML = '<p style="text-align: center; color: #666; padding: 2rem;">No complaints in system.</p>';
                return;
            }

            let html = '<table class="status-table"><thead><tr><th>ID</th><th>Student</th><th>Room</th><th>Type</th><th>Date</th><th>Status</th><th>Actions</th></tr></thead><tbody>';
            
            complaints.forEach(complaint => {
                const statusClass = complaint.status === 'Resolved' ? 'status-resolved' : 
                                   complaint.status === 'In Progress' ? 'status-progress' : 'status-pending';
                html += `
                    <tr>
                        <td>${complaint.id}</td>
                        <td>${complaint.studentName}</td>
                        <td>${complaint.roomNo}</td>
                        <td>${complaint.type}</td>
                        <td>${complaint.date}</td>
                        <td><span class="status-badge ${statusClass}">${complaint.status}</span></td>
                        <td>
                            <div class="action-buttons">
                                <button class="btn-small btn-success" onclick="updateStatus('${complaint.id}', 'In Progress')">Progress</button>
                                <button class="btn-small btn-success" onclick="updateStatus('${complaint.id}', 'Resolved')">Resolve</button>
                            </div>
                        </td>
                    </tr>
                `;
            });
            
            html += '</tbody></table>';
            container.innerHTML = html;
        }

        // Update complaint status
        function updateStatus(complaintId, newStatus) {
            const complaint = complaints.find(c => c.id === complaintId);
            if (complaint) {
                complaint.status = newStatus;
                displayAdminDashboard();
                alert(`Complaint ${complaintId} status updated to ${newStatus}`);
            }
        }

        // Initialize on load
        initSystem();
    </script>
